\subsection{Stacking Ensemble Model}
% (~350 words + Figure 2)
\textbf{[TODO: Write subsection III-C intro paragraph]}

% Mathematical formulation
Let $B_1$, $B_2$, $B_3$ denote the base learners (Random Forest, XGBoost, LightGBM) respectively. For an input feature vector $\mathbf{x}$, the base learner probability outputs are:
\begin{equation}
    P_i = B_i(\mathbf{x}), \quad i \in \{1, 2, 3\}
    \label{eq:base_predictions}
\end{equation}
where $P_i \in \mathbb{R}^{C}$ is a probability vector over $C = 22$ crop classes. The meta-learner input is formed by concatenation:
\begin{equation}
    \mathbf{Z} = [P_1 \oplus P_2 \oplus P_3] \in \mathbb{R}^{3C}
    \label{eq:meta_input}
\end{equation}
The final prediction is computed by the Logistic Regression meta-learner:
\begin{equation}
    \hat{y} = \text{Meta-Learner}(\mathbf{Z}) = \arg\max_{c} \; \sigma(\mathbf{W}\mathbf{Z} + \mathbf{b})_c
    \label{eq:final_pred}
\end{equation}
where $\mathbf{W}$ and $\mathbf{b}$ are learned weight matrix and bias vector, and $\sigma(\cdot)$ denotes the softmax function.

\subsection{SHAP-Based Explainability}
% (~250 words)
% Background: SHAP (SHapley Additive exPlanations) based on cooperative game theory.
% The additive explanation model:
\textbf{[TODO: Write subsection III-D intro]}

The SHAP explanation for a prediction is defined as:
\begin{equation}
    f(\mathbf{x}) = \phi_0 + \sum_{j=1}^{M} \phi_j
    \label{eq:shap}
\end{equation}
where $f(\mathbf{x})$ is the model output, $\phi_0$ is the base value (average model output), $\phi_j$ is the SHAP value of feature $j$, and $M$ is the total number of features.

The Return on Investment (ROI) for a recommended crop is calculated as:
\begin{equation}
    \text{ROI} = \frac{(\text{Expected Yield} \times \text{Market Price}) - \text{Total Input Cost}}{\text{Total Input Cost}} \times 100\%
    \label{eq:roi}
\end{equation}
